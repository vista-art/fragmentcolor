## Description

The [TextureTarget](/api/targets/texturetarget) is an offscreen [Target](/api/core/target) backed by a GPU texture.

Use it for headless rendering, tests, server-side image generation, or CI.



## Example

<Tabs>

<TabItem label="Rust">
<Code
code={`
async fn run() -> Result<(), Box<dyn std::error::Error>> {

use fragmentcolor::{Renderer, Shader, Target};

let renderer = Renderer::new();
let mut target = renderer.create_texture_target([64, 64]).await?;

let shader = Shader::default();
renderer.render(&shader, &target)?;

let image = target.get_image();

assert_eq!(image.len(), 64 * 64 * 4); // RGBA8
target.resize([128, 128]);
assert_eq!(target.size().width, 128);
assert_eq!(target.size().height, 128);
Ok(())
}
fn main() -> Result<(), Box<dyn std::error::Error>> { pollster::block_on(run()) }
`}

### TextureTarget::size() -&gt; Size &#123; width: u32, height: u32, depth: u32 &#125;

Returns the size of the [TextureTarget](/api/targets/texturetarget) in pixels.

#### Example

<Tabs>

<TabItem label="Rust">
<Code
code={`
async fn run() -> Result<(), Box<dyn std::error::Error>> {

use fragmentcolor::Renderer;

let renderer = Renderer::new();
let target = renderer.create_texture_target([64, 64]).await?;
let size = target.size();
let width = size.width;
let height = size.height;
let depth = size.depth;

assert_eq!(target.size(), [64, 64]);
assert_eq!(width, 64);
assert_eq!(height, 64);
assert_eq!(depth, 1);
Ok(())
}
